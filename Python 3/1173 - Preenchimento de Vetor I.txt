# Preenchimento de vetores I

v = int(input())
vetor = [v]

for x in range(1, 10):
    vetor.append((vetor[x - 1]) + (vetor[x - 1]))

for y in range(len(vetor)):
    print('N[{}] = {}'.format(y, vetor[y]))
