# Soma de Ímpares Consecutivos I

n1 = int(input())
n2 = int(input())
soma = 0

if n1 >= n2:
    n1, n2 = n2, n1

while n1 < n2 - 1:
    n1 = n1 + 1
    if n1 % 2 != 0:
        soma = soma + n1
print(soma)
