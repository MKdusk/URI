# Fibonacci Fácil

n = int(input())
fi = [0, 1]
soma = 0
for x in range(2, n):
        soma = fi[x - 2] + fi[x - 1]
        fi.append(soma)

for y in range(len(fi) - 1):
    print(fi[y], end=' ')
print(fi[len(fi) - 1])
