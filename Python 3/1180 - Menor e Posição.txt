# Menor e Posição

n = int(input())
x = []
numeros = str(input()).split()
menor = 1001
posicao = 0

for y in range(n):
    x.append(int(numeros[y]))

for w in range(len(x)):
    if x[w] < menor:
        menor = x[w]
        posicao = w

print('Menor valor: {}'.format(menor))
print('Posicao: {}'.format(posicao))
