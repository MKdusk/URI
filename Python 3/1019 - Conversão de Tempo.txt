#Conversão de tempo
horaS = 3600
minutoS = 60

tempoS = int(input())

hora = tempoS // horaS

tempoS = tempoS - (hora * horaS)

minuto = tempoS // minutoS

tempoS = tempoS - (minuto * minutoS)

print('{}:{}:{}'.format(hora,minuto,tempoS))
