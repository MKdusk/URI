# Pares entre Cinco Números
numeros = []
cont = 0

for x in range(5):
    a = float(input())
    numeros.append(a)

for x in range(len(numeros)):
    if numeros[x] % 2 == 0:
        cont = cont + 1

print('{} valores pares'.format(cont))
