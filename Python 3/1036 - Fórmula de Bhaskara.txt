# Fórmula de Bhaskara
texto = str(input())
textoq = texto.split()
a = float(textoq[0])
b = float(textoq[1])
c = float(textoq[2])

delta = ((b**2) - 4 * a * c)

if delta == 0 or delta < 0 or a == 0:
    print('Impossivel calcular')
else:
    r1 = (-b + delta**(1/2)) / (2 * a)
    r2 = (-b - delta**(1/2)) / (2 * a)
    print('R1 = {:.5f}'.format(r1))
    print('R2 = {:.5f}'.format(r2))
